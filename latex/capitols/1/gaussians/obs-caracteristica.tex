\begin{obs}
Si $X \sim N(0,t)$, la seva funció característica és
\begin{equation*}
\varphi_X(u)
= \mathbb{E}[e^{iuX}] 
= \frac{1}{\sqrt{2\pi t}} \int_{\mathbb{R}} e^{iux} e^{-\frac{x^2}{2t}} \, dx
= \frac{e^{-u^2 t/2}}{\sqrt{2\pi t}} \int_{\mathbb{R}} e^{-\frac{(x - i u t)^2}{2t}} \, dx
= e^{-\frac{u^2 t}{2}}.
\end{equation*}
\end{obs}
