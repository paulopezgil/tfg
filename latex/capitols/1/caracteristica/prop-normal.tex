\begin{prop} \label{prop:caracteristica-normal}
Si $X \sim \mathcal{N}(0,t)$, la seva funció característica és
\begin{equation*}
\varphi_X(u) = e^{-\frac{u^2 t}{2}},
\end{equation*}
\begin{prova}
Per definició, tenim que
\begin{equation*}
    \varphi_X(u)
    = \mathbb{E}[e^{iuX}] 
    = \frac{1}{\sqrt{2\pi t}} \int_{\mathbb{R}} e^{iux} e^{-\frac{x^2}{2t}} \, dx.
\end{equation*}
Completant el quadrat en l'exponent, obtenim
\begin{equation*}
    \varphi_X(u)
    = \frac{e^{-u^2 t/2}}{\sqrt{2\pi t}} \int_{\mathbb{R}} e^{-\frac{(x - i u t)^2}{2t}} \, dx
    = e^{-\frac{u^2 t}{2}}.
\end{equation*}
\end{prova}
\end{prop}
