\begin{defi}
Considerem un espai de probabilitat $(\Omega, \mathcal{F}, \P)$,
un espai mesurable $(S, \mathcal{S})$
i una funció $X:\Omega\to S$.
Diem que $X$ és un \emph{element aleatori} si és $(\mathcal{F}/\mathcal{S})$-mesurable,
és a dir, si $X^{-1}(B) \in \mathcal{F}$ per a tot $B \in \mathcal{S}$. En particular,
\begin{itemize}
\item Diem que $X$ és una \emph{variable aleatòria} si $S=\mathbb{R}$.
\item Diem que $X$ és un \emph{vector aleatòri} si $S=\mathbb{R}^k$, $k > 1$.
\item Diem que $X$ és una \emph{funció aleatòria} si $S=C[0,\infty]$ o algun altre espai de funcions.
\end{itemize}
\end{defi}
