\begin{prop} \label{obs-relativament-compacte}
Si una successió $\left\{\P_n\right\}_n \subset \mathcal{P}(S)$ de mesures de probabilitat
convergeix feblement cap a una mesura de probabilitat $\P$,
aleshores és relativament compacte.
\begin{prova}
Sigui $\{\P_{n_k}\}_k \subset \{\P_n\}_n$ una subsuccessió qualsevol.
Aleshores, o bé $\P_{n_k} \xrightarrow{w} \P$,
o bé existeix un $k_0 \in \mathbb{N}$ tal que
$\P_{n_k} = \P_{n_{k_0}}$ per a a tot $k > k_0$, i per tant
$\P_{n_k} \xrightarrow{w} \P_{n_{k_0}}$.
\end{prova}
\end{prop}
