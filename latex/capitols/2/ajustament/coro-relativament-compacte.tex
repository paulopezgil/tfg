\begin{corol} \label{coro-relativament-compacte}
Si una successió de variables aleatòries $\{X_n\}_n$
convergeix en llei cap a una variable aleatòria $X$,
aleshores és relativament compacte.
\begin{prova}
Per definició, tenim que $\mathcal{L}(X_n) \xrightarrow{w} \mathcal{L}(X)$.
Per la Proposició \ref{obs-relativament-compacte}, aleshores
$\{\mathcal{L}(X_n)\}_n$ és relativament compacte.
\end{prova}
\end{corol}
