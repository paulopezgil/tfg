\begin{corol} \label{coro-relativament-compacte}
Si una successió de variables aleatòries $\{X_n\}_n$
convergeix en llei cap a una variable aleatòria $X$,
aleshores és relativament compacta.
\begin{prova}
Per definició, $\mathcal{L}(X_n) \xrightarrow{w} \mathcal{L}(X)$.
Per tant, $\{\mathcal{L}(X_n)\}_n$ és relativament compacta.
\end{prova}
\end{corol}
