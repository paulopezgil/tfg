\begin{lema} \label{lema-mapa-continua}
Siguin $\left\{ X^{(n)} \right\}_n$ i $X$ elements aleatoris
amb valors a $\left(S, \mathcal{B}(S)\right)$.
Sigui $\left(S', \mathcal{B}(S')\right)$ un altre espai mètric
i sigui $\varphi: S \to S'$ una funció contínua.
Si $X^{(n)} \xrightarrow{\mathcal{L}} X$, aleshores
$\varphi(X^{(n)}) \xrightarrow{\mathcal{L}} \varphi(X)$.

\begin{prova}
Sigui $f: S' \to \mathbb{R}$ una funció contínua i acotada.
Aleshores, $f \circ \varphi: S \to \mathbb{R}$ és també contínua i acotada.
Per l'observació \ref{obs-convergencia-en-llei},
com $X^{(n)} \xrightarrow{\mathcal{L}} X$, aleshores
\begin{equation*}
\lim_{n \to \infty} \mathbb{E}[f(\varphi(X^{(n)}))] =
\lim_{n \to \infty} \mathbb{E}[(f \circ \varphi)(X^{(n)})] =
\mathbb{E}[(f \circ \varphi)(X)] = \mathbb{E}[f(\varphi(X))].
\end{equation*}
Novament per l'observació \ref{obs-convergencia-en-llei}, això equival a
$\varphi(X^{(n)}) \xrightarrow{\mathcal{L}} \varphi(X)$.
\end{prova}
\end{lema}
